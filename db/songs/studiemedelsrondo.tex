\begin{song}

\songtitle{Studiemedelsrondo}
\firstline{Vi dricker punsch}
\begin{songmeta}
Melodi: Vi lossar sand\ldots
\end{songmeta}

\begin{songtext}
Vi dricker punsch
till lunch,
när vi har fått avin.
Vi lunchar hela dagen,
tills kassan gått i sin.

För mycket punsch
till lunch,
nu mår jag som ett svin.
Helkass hela dagen,
och munnen en latrin.
\end{songtext}
\end{song}
