\begin{song}
\songtitle{Ju mera öl vi dricker}

\begin{songmeta}
Melodi: Ju mer vi är tillsammans
\end{songmeta}

\begin{songtext}
Ju mera öl vi dricker,
vi dricker, vi dricker,
ju mera öl vi dricker,
ju rundare vi bli.

För rundare är sundare
och sundare är rundare.
Ju mera öl vi dricker,
ju rundare vi bli.
\end{songtext}
\end{song}
