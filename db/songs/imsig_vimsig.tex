\begin{song}
\songtitle{Imsig vimsig}

\begin{songmeta}
Melodi: Imse vimse spindel
\end{songmeta}

\begin{songtext}
Imsig, vimsig blir man utav lite vin,
klättrar upp på stolen, verkar piggelin,
ramlar under bordet, sussar en minut.
Vaknar av att vinet i glaset tagit slut.
\end{songtext}
\end{song}
