\begin{song}
\songtitle{Minnet}
\firstline{Minne? Muisti hävis, mutt' minne}

\begin{songmeta}
Melodi: Memory (Andrew Lloyd Webber)
Text: Bosse Österberg
Språk: Finska
\end{songmeta}

\begin{songtext}
Minne? Muisti hävis, mutt' minne?
Juhlista selvisimme
muistikatkoja on.

Minne, lähtisin vaikka minne,
kunhan selvittäisimme
mitä tapahtunut on.

Mutta tiedän
mä keinon mikä auttaapi tuo:
Ota ryyppy, ja muistis juo!
\end{songtext}

\begin{songnotes}
Teknologföreningen, Helsingfors.
\end{songnotes}
\end{song}
