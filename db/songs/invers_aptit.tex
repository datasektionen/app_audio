\begin{song}

\songtitle{Invers aptit}
\firstline{Nu fyllas många magar små}
\begin{songmeta}
Melodi: Nu tändas tusen juleljus, Sv. psalm 116
Musik: Emmy Köhler
\end{songmeta}

\begin{songtext}
Nu fyllas många magar små
av iskall renad sprit.
Men några kastar åter opp.
Det är invers aptit.
\end{songtext}
\end{song}
